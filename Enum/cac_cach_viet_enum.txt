typedef enum logic [1:0] {
    IDLE,
    START,
    RUN,
    STOP
} state_t;

state_t current_state, next_state;

always_ff @(posedge clk)
    current_state <= next_state;

always_comb begin

    case(current_state)

        IDLE  : next_state = START;
        START : next_state = RUN;
        RUN   : next_state = STOP;
        STOP  : next_state = IDLE;

        default: next_state = IDLE;

    endcase

end
