1. typedef struct packed

typedef struct packed {
    logic [31:0] addr;
    logic [31:0] data;
    logic        valid;
} packet_t;

packet_t pkt;


2. Struct Initialization

// Named assignment
packet_t pkt = '{
    addr  : 32'h1000,
    data  : 32'hDEADBEEF,
    valid : 1'b1
};

// Default assignment
packet_t pkt = '{default:0};

// Positional assignment
packet_t pkt = '{32'h1000, 32'hDEADBEEF, 1'b1};


3. Access Struct Members

pkt.addr
pkt.data
pkt.valid


4. Struct Assignment

packet_t pkt1, pkt2;

pkt2 = pkt1;


5. Array of Structs

typedef struct packed {
    logic [7:0]  addr;

    logic        valid;
} entry_t;

entry_t memory [16];

memory[0].addr
memory[5].data
memory[10].valid


6. Nested Struct

typedef struct packed {
    logic [7:0]  day;
    logic [7:0]  month;
    logic [15:0] year;
} date_t;

typedef struct packed {
    logic [31:0] id;
    date_t birthday;
} employee_t;

employee.birthday.year

    IDLE,
    RUN,
    STOP
} state_t;

typedef struct packed {
    logic [31:0] data;

} packet_t;
